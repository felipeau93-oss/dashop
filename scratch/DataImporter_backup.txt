import React, { useState } from 'react';
import Papa from 'papaparse';
import { UploadCloud, FileSpreadsheet, CheckCircle2, AlertTriangle, ArrowRight, Loader2, Database, Box, DollarSign } from 'lucide-react';
import * as XLSX from 'xlsx';

export default function DataImporter({ onImportOperacional, onImportBilling, onImportCapCar, onImportOperacionalBSC, rawFaturamentoData = [], rawOperacionalData = [], mapeamentoFiliais = [] }) {
  // === ESTADO COMPARTILHADO ===
  const [logs, setLogs] = useState([]);
  const [activeStep, setActiveStep] = useState(1); // 1 = Operacional, 2 = Billing, 3 = CAP, 4 = BSC

  // === ESTADO ETAPA 1: OPERACIONAL ===
  const [baseFile, setBaseFile] = useState(null);
  const [quinzenaOp, setQuinzenaOp] = useState('');
  const [isProcessingOp, setIsProcessingOp] = useState(false);
  const [previewOp, setPreviewOp] = useState(null);
  const [isSavingOp, setIsSavingOp] = useState(false);

  // === ESTADO ETAPA 2: BILLING ===
  const [billingFile, setBillingFile] = useState(null);
  const [quinzenaBilling, setQuinzenaBilling] = useState('');
  const [numeroFaturaBilling, setNumeroFaturaBilling] = useState('');
  const [isProcessingBilling, setIsProcessingBilling] = useState(false);
  const [previewBilling, setPreviewBilling] = useState(null);
  const [isSavingBilling, setIsSavingBilling] = useState(false);

  // === ESTADO ETAPA 3: CAP ===
  const [capcarFile, setCapcarFile] = useState(null);
  const [isProcessingCapCar, setIsProcessingCapCar] = useState(false);
  const [previewCapCar, setPreviewCapCar] = useState(null);
  const [isSavingCapCar, setIsSavingCapCar] = useState(false);

  // === ESTADO ETAPA 4: OPERACIONAL BSC ===
  const [bscFile, setBscFile] = useState(null);
  const [isProcessingBsc, setIsProcessingBsc] = useState(false);
  const [previewBsc, setPreviewBsc] = useState(null);
  const [isSavingBsc, setIsSavingBsc] = useState(false);

  const addLog = (msg, type = 'info') => {
    setLogs(prev => [...prev, { time: new Date().toLocaleTimeString(), msg, type }]);
  };

  // ============================================================================
  // ETAPA 1: PROCESSAR OPERACIONAL
  // ============================================================================
  const handleProcessOperacional = () => {
    if (!baseFile) return;
    setIsProcessingOp(true);
    setLogs(prev => [...prev]);
    setPreviewOp(null);
    addLog('[ETAPA 1] Iniciando leitura da Base Operacional...', 'info');

    const processBaseFile = (dataArray) => {
      let headerRowIdx = -1;
      let rotaColIdx = -1;
      let xptColIdx = -1;
      let saldoColIdx = -1;
      let entreguesColIdx = -1;
      let insucessosColIdx = -1;
      let driverIdColIdx = -1;
      let motoristaColIdx = -1;

      for (let i = 0; i < Math.min(15, dataArray.length); i++) {
        const row = dataArray[i];
        if (!row) continue;
        const cRota = row.findIndex(c => String(c).toUpperCase().includes('ROTA'));
        const cXpt = row.findIndex(c => String(c).toUpperCase() === 'XPT' || String(c).toUpperCase() === 'FILIAL');
        const cSaldo = row.findIndex(c => String(c).toUpperCase() === 'SALDO');
        const cEntregues = row.findIndex(c => String(c).toUpperCase() === 'ENTREGUES');
        const cInsucessos = row.findIndex(c => String(c).toUpperCase() === 'INSUCESSOS');
        const cDriver = row.findIndex((c, idx) => String(c).toUpperCase() === 'DRIVER' || String(c).toUpperCase() === 'DRIVER ID' || (String(c).trim() === '#' && idx === 5));
        const cMotorista = row.findIndex(c => String(c).toUpperCase() === 'MOTORISTA' || String(c).toUpperCase() === 'DRIVER NAME' || String(c).toUpperCase() === 'NOME');

        if (cRota !== -1 && cXpt !== -1) {
          headerRowIdx = i;
          rotaColIdx = cRota;
          xptColIdx = cXpt;
          saldoColIdx = cSaldo;
          entreguesColIdx = cEntregues;
          insucessosColIdx = cInsucessos;
          driverIdColIdx = cDriver;
          motoristaColIdx = cMotorista;
          break;
        }
      }

      if (headerRowIdx === -1) {
        addLog('Erro: Não encontrou colunas ROTA e XPT/FILIAL na planilha.', 'error');
        setIsProcessingOp(false);
        return;
      }

      const configMap = {};
      mapeamentoFiliais.forEach(m => configMap[String(m.filial).toUpperCase()] = m);

      const newOperacionalData = [];
      const mapRotaFilial = {};
      let insucessosHeaders = [];

      const mapInsucessosOp = {
        'QTD BLOQUEADO': 'Bloqueado',
        'QTD BLOQUEADO POR PALAVRA': 'Palavra Chave',
        'QTD CANCELADO': 'Cancelado',
        'QTD COLETADO': 'Coleta',
        'QTD COMÉRCIO FECHADO': 'Comercial',
        'QTD COMPRADOR REJEITOU': 'Recusa',
        'QTD DANIFICADO': 'Avaria',
        'QTD DESTINATÁRIO MUDOU': 'Mudou Endereco',
        'QTD ENDEREÇO INACESSÍVEL': 'Area Inacessivel',
        'QTD ENDEREÇO NÃO VISITADO': 'Nao Visitado',
        'QTD ENDEREÇO RUIM': 'Nao Localizado',
        'QTD FALTANDO': 'Faltante',
        'QTD FORA DE ROTA': 'Fora de Rota',
        'QTD NINGUÉM PRA RECEBER': 'Cliente Ausente',
        'QTD TENTATIVA DE ROUBO': 'Tentativa Roubo',
        'BLOQUEADO': 'Bloqueado',
        'PALAVRA CHAVE': 'Palavra Chave',
        'CANCELADO': 'Cancelado',
        'COLETA': 'Coleta',
        'COMERCIAL': 'Comercial',
        'RECUSA': 'Recusa',
        'AVARIA': 'Avaria',
        'MUDOU ENDERECO': 'Mudou Endereco',
        'AREA INACESSIVEL': 'Area Inacessivel',
        'NAO VISITADO': 'Nao Visitado',
        'NAO LOCALIZADO': 'Nao Localizado',
        'FALTANTE': 'Faltante',
        'FORA DE ROTA': 'Fora de Rota',
        'CLIENTE AUSENTE': 'Cliente Ausente',
        'TENTATIVA ROUBO': 'Tentativa Roubo'
      };

      const normalizeText = (text) => {
        return String(text)
          .normalize('NFD')
          .replace(/[\u0300-\u036f]/g, '')
          .trim()
          .toUpperCase();
      };

      const row = dataArray[headerRowIdx];
      const rawHeadersOp = row.map(h => String(h || '').trim());

      for (let j = 0; j < row.length; j++) {
        const colName = normalizeText(row[j]);
        if (mapInsucessosOp[colName]) {
          insucessosHeaders.push({ index: j, name: mapInsucessosOp[colName] });
        }
      }

      // Fallback caso não ache nas mapeadas e tenha a coluna INSUCESSOS
      if (insucessosHeaders.length === 0 && insucessosColIdx !== -1) {
        for (let j = insucessosColIdx + 1; j < row.length; j++) {
          if (row[j] && String(row[j]).trim() !== '') {
            insucessosHeaders.push({ index: j, name: String(row[j]).trim() });
          }
        }
      }
      
      const invalidOpKeys = ['% evid micro', '% evid distância', 'não rateadas', 'contar % entregue', '% evidenciado', '% entregue', 'km percorrido', 'paradas com adicional', 'total da taxa de entrega', 'volume'];
      insucessosHeaders = insucessosHeaders.filter(h => !invalidOpKeys.some(inv => h.name.toLowerCase().includes(inv)));

      for (let i = headerRowIdx + 1; i < dataArray.length; i++) {
        const row = dataArray[i];
        if (!row) continue;
        const rota = String(row[rotaColIdx] || '').trim();
        const filial = String(row[xptColIdx] || '').trim();

        if (rota && filial && filial !== 'N/A') {
          const fKey = filial.toUpperCase();
          const config = configMap[fKey] || {};
          const regional = config.regional || 'N/A';
          const supervisor = config.supervisor || 'N/A';

          mapRotaFilial[rota] = { filial, regional, supervisor };

          const saldo = saldoColIdx !== -1 ? (parseFloat(row[saldoColIdx]) || 0) : 0;
          const entregues = entreguesColIdx !== -1 ? (parseFloat(row[entreguesColIdx]) || 0) : 0;
          const driverId = driverIdColIdx !== -1 ? String(row[driverIdColIdx]).trim() : '';
          const motorista = motoristaColIdx !== -1 ? String(row[motoristaColIdx]).trim() : 'N/A';

          const dados_originais = {};
          rawHeadersOp.forEach((h, idx) => {
            if (h) dados_originais[h] = row[idx];
          });

          const insucessosDetalhados = {};
          insucessosHeaders.forEach(h => {
            const v = parseFloat(row[h.index]) || 0;
            if (v > 0) insucessosDetalhados[h.name] = v;
          });

          const somaIns = Object.values(insucessosDetalhados).reduce((a,b)=>a+b, 0);
          if (saldo > 0 || entregues > 0 || somaIns > 0 || driverId) {
            newOperacionalData.push({
              id_rota: rota,
              filial,
              regional,
              supervisor,
              saldo,
              entregues,
              insucessosDetalhados,
              quinzena: quinzenaOp.trim() || 'N/A',
              dia_semana: 'N/A',
              cluster: 'N/A',
              driver_id: driverId,
              motorista: motorista || 'N/A',
              dados_originais
            });
          }
        }
      }

      const totalRotas = Object.keys(mapRotaFilial).length;
      addLog(`Mapa PROCV extraído: ${totalRotas} rotas mapeadas para filiais.`, 'success');
      addLog(`Dados Operacionais: ${newOperacionalData.length} linhas com volumetria (Saldo/Entregues).`, 'success');

      setPreviewOp({
        operacional: newOperacionalData,
        totalRotas,
        mapRotaFilial
      });
      setIsProcessingOp(false);
    };

    const ext = baseFile.name.split('.').pop().toLowerCase();
    if (ext === 'csv') {
      Papa.parse(baseFile, {
        skipEmptyLines: true,
        complete: (baseResults) => processBaseFile(baseResults.data),
        error: (err) => {
          addLog(`Erro ao ler Base Operacional: ${err.message}`, 'error');
          setIsProcessingOp(false);
        }
      });
    } else {
      const reader = new FileReader();
      reader.onload = (e) => {
        try {
          const data = new Uint8Array(e.target.result);
          const workbook = XLSX.read(data, { type: 'array' });
          const firstSheetName = workbook.SheetNames[0];
          const worksheet = workbook.Sheets[firstSheetName];
          const jsonData = XLSX.utils.sheet_to_json(worksheet, { header: 1 });
          processBaseFile(jsonData);
        } catch (err) {
          addLog(`Erro ao ler arquivo Excel: ${err.message}`, 'error');
          setIsProcessingOp(false);
        }
      };
      reader.readAsArrayBuffer(baseFile);
    }
  };

  const handleSaveOperacional = async () => {
    setIsSavingOp(true);
    addLog('[ETAPA 1] Salvando Operacional no Firebase...', 'info');
    try {
      if (onImportOperacional) {
        await onImportOperacional(previewOp.operacional);
      }
      addLog('[ETAPA 1] Operacional salvo com sucesso! Você pode prosseguir para o Billing ou importar mais dados.', 'success');
      setPreviewOp(null);
      setBaseFile(null);
      setQuinzenaOp('');
    } catch (err) {
      addLog(`Erro ao salvar operacional: ${err.message}`, 'error');
    } finally {
      setIsSavingOp(false);
    }
  };

  // ============================================================================
  // ETAPA 2: PROCESSAR BILLING
  // ============================================================================
  const handleProcessBilling = () => {
    if (!billingFile) return;
    setIsProcessingBilling(true);
    setLogs(prev => [...prev]);
    setPreviewBilling(null);
    addLog('[ETAPA 2] Iniciando processamento do Billing...');

    const mapRotaFilial = {};
    const configMap = {};
    mapeamentoFiliais.forEach(m => configMap[String(m.filial).toUpperCase()] = m);

    let totalRotasConhecidas = 0;
    const populate = (d) => {
      if (d.id_rota && d.filial && d.filial !== 'N/A') {
        const fKey = String(d.filial).toUpperCase();
        const config = configMap[fKey] || {};
        mapRotaFilial[String(d.id_rota).trim()] = {
          filial: String(d.filial).trim(),
          regional: config.regional || 'N/A',
          supervisor: config.supervisor || 'N/A'
        };
        totalRotasConhecidas++;
      }
    };
    rawFaturamentoData.forEach(populate);
    rawOperacionalData.forEach(populate);

    addLog(`Base de cruzamento: ${Object.keys(mapRotaFilial).length} rotas únicas conhecidas (de ${totalRotasConhecidas} registros no Firebase).`, 'info');

    Papa.parse(billingFile, {
      skipEmptyLines: true,
      complete: (billingResults) => {
        processBilling(billingResults.data, mapRotaFilial, quinzenaBilling.trim(), numeroFaturaBilling.trim());
      },
      error: (err) => {
        addLog(`Erro ao ler Billing: ${err.message}`, 'error');
        setIsProcessingBilling(false);
      }
    });
  };

  const processBilling = (data, mapRotaFilial, quinzenaBillingManual, numeroFatura) => {
    addLog(`Billing lido: ${data.length} linhas brutas.`);

    let headerIdx = -1;
    let quinzenaStr = 'N/A';

    for (let i = 0; i < Math.min(20, data.length); i++) {
      const row = data[i];
      if (!quinzenaBillingManual && row.some(c => String(c).toUpperCase().includes('PERÍODO') || String(c).toUpperCase().includes('PERIODO'))) {
        if (data[i+1] && data[i+1].length > 5) {
          quinzenaStr = String(data[i+1][5] || data[i+1][4] || '').trim();
        }
      }
      
      const descIdx = row.findIndex(c => String(c).toUpperCase() === 'DESCRIÇÃO' || String(c).toUpperCase() === 'DESCRICAO');
      if (descIdx !== -1) {
        headerIdx = i;
        break;
      }
    }

    if (headerIdx === -1) {
      addLog(`Erro: Coluna 'Descrição' não encontrada no Billing.`, 'error');
      setIsProcessingBilling(false);
      return;
    }

    if (quinzenaBillingManual) {
      quinzenaStr = quinzenaBillingManual;
    }

    addLog(`Quinzena identificada para o Billing: ${quinzenaStr}`);

    const rawHeadersBilling = data[headerIdx].map(h => String(h || '').trim());

    let isPenaltiesSection = false;
    const mapDiarias = {};
    const arrayPenalidades = [];

    let linhasApagadas = 0;
    let acentosRemovidos = 0;

    for (let i = headerIdx + 1; i < data.length; i++) {
      const row = data[i];
      const rowText = row.join('').trim().toLowerCase();

      const dados_originais = {};
      rawHeadersBilling.forEach((h, idx) => {
        if (h) dados_originais[h] = row[idx];
      });

      let parseCurrencyString = (str) => {
        let p = String(str || '0').toUpperCase().replace('R$', '').trim();
        if (p.includes('.') && p.includes(',')) {
          let lastDot = p.lastIndexOf('.');
          let lastComma = p.lastIndexOf(',');
          if (lastComma > lastDot) p = p.replace(/\./g, '').replace(',', '.');
          else p = p.replace(/,/g, '');
        } else if (p.includes(',')) {
          p = p.replace(',', '.');
        }
        return parseFloat(p) || 0;
      };

      if (rowText === '' || rowText.includes('total') || rowText.includes('subtotal') || rowText.includes('faturação')) {
        linhasApagadas++;
        continue;
      }

      let descricaoOrig = String(row[0] || '');
      let idRota = String(row[1] || '').trim();
      let dataInicio = String(row[2] || '').trim();
      let dataTermino = String(row[3] || '').trim();
      let placa = String(row[4] || '').trim();
      let motorista = String(row[5] || '').trim();
      let quantidade = parseFloat(String(row[6] || '0').replace(',', '.')) || 0;
      let precoUnitario = parseCurrencyString(row[7]);
      let totalValue = parseCurrencyString(row[8] || row[7]); // Fallback se não tiver Total

      let descLimpa = descricaoOrig.normalize("NFD").replace(/[\u0300-\u036f]/g, "");
      if (descricaoOrig !== descLimpa) acentosRemovidos++;
      
      descLimpa = descLimpa.replace(/Ve[ií]culo de Passeio/gi, "Passeio");
      descLimpa = descLimpa.replace(/Utilit[aá]rios/gi, "Utilitario");

      const routeMapObj = mapRotaFilial[idRota] || { filial: 'N/A', regional: 'N/A', supervisor: 'N/A' };
      const filialProc = routeMapObj.filial;
      const regionalProc = routeMapObj.regional;
      const supervisorProc = routeMapObj.supervisor;

      let textoUpper = descLimpa.toUpperCase();
      let dLower = descLimpa.toLowerCase();

      if (dLower.includes('visited addresses') && !dLower.includes('not visited')) {
         if (!idRota) idRota = `sem_rota_${i}`;
         if (!mapDiarias[idRota]) {
            mapDiarias[idRota] = {
              quinzena: quinzenaStr, numero_fatura: numeroFatura, filial: filialProc, regional: regionalProc, supervisor: supervisorProc,
              descricao_original: descLimpa, id_rota: idRota, data_inicio: dataInicio, placa: placa, motorista: motorista, faturamento: 0, faturamento_paradas: 0, qtd_paradas: 0, svc: "", tipo: "", ciclo: "", range_km: "", dia_semana: "", categoria: "", dados_originais
            };
         }
         mapDiarias[idRota].faturamento_paradas += totalValue;
         mapDiarias[idRota].qtd_paradas += quantidade;
         mapDiarias[idRota].faturamento += totalValue; // Soma as paradas no faturamento total

         isPenaltiesSection = true;
         continue;
      }

      if (!isPenaltiesSection) {
        let isAddStops = textoUpper.includes("ADDSTOPS");
        if (!idRota) idRota = `sem_rota_${i}`;

        if (!mapDiarias[idRota]) {
          mapDiarias[idRota] = {
            quinzena: quinzenaStr, numero_fatura: numeroFatura, filial: filialProc, regional: regionalProc, supervisor: supervisorProc,
            descricao_original: descLimpa, id_rota: idRota, data_inicio: dataInicio, placa: placa, motorista: motorista, faturamento: 0, faturamento_paradas: 0, qtd_paradas: 0, svc: "", tipo: "", ciclo: "", range_km: "", dia_semana: "", categoria: "", dados_originais
          };
        }

        if (isAddStops) {
          mapDiarias[idRota].faturamento_paradas += totalValue;
          mapDiarias[idRota].qtd_paradas += quantidade;
          mapDiarias[idRota].faturamento += totalValue;
        } else {
          mapDiarias[idRota].faturamento += precoUnitario;

          let categoria = descLimpa.split(/ - SVC:/i)[0].trim();
          let matchSvc = descLimpa.match(/SVC:\s*([A-Z0-9]+)/i);
          let svc = matchSvc ? matchSvc[1] : "";
          let tipoDD = textoUpper.includes("AMBULANCE") ? "AMBULANCE" : "NORMAL";
          let ciclo = textoUpper.includes("PART TIME") ? "PM" : "AM";
          let matchRange = descLimpa.match(/KMs RANGE:\s*([\d\/]+)/i);
          let range = matchRange ? matchRange[1] : "1/100";
          let diaSem = textoUpper.includes("HOLIDAY") ? "Dom-Fer" : "Seg-Sab";

          mapDiarias[idRota].descricao_original = descLimpa;
          mapDiarias[idRota].categoria = categoria;
          mapDiarias[idRota].svc = svc;
          mapDiarias[idRota].tipo = tipoDD;
          mapDiarias[idRota].ciclo = ciclo;
          mapDiarias[idRota].range_km = range;
          mapDiarias[idRota].dia_semana = diaSem;
          if (dataInicio) mapDiarias[idRota].data_inicio = dataInicio;
          if (placa) mapDiarias[idRota].placa = placa;
          if (motorista) mapDiarias[idRota].motorista = motorista;
        }

      } else {
        let idPacote = "";
        let tipoPenalidade = "Outros";

        if (dLower.includes("pnr")) tipoPenalidade = "PNRs";
        else if (dLower.includes("lost package") || dLower.includes("extravio")) tipoPenalidade = "Lost Packages";
        else if (dLower.includes("vehicle daily not visited") || dLower.includes("falta")) tipoPenalidade = "Not Visited";

        if (tipoPenalidade !== "Not Visited") {
          let matchID = descLimpa.match(/(\d+)\s*$/);
          idPacote = matchID ? matchID[1] : "";
        }

        arrayPenalidades.push({
          quinzena: quinzenaStr,
          numero_fatura: numeroFatura,
          filial: filialProc,
          id_pacote: idPacote,
          descricao: descLimpa,
          tipo: tipoPenalidade,
          id_rota: idRota,
          data: dataInicio,
          placa: placa,
          motorista: motorista,
          valor: Math.abs(totalValue),
          dados_originais
        });
      }
    }

    const arrayDiarias = Object.values(mapDiarias);

    addLog(`Limpeza: ${linhasApagadas} linhas removidas (Totais/Vazias), ${acentosRemovidos} acentos retirados.`, 'success');
    addLog(`Diárias (Rotas Únicas) processadas: ${arrayDiarias.length}`, 'success');
    addLog(`Penalidades processadas: ${arrayPenalidades.length}`, 'success');

    const rotasNaoEncontradas = new Set();
    arrayDiarias.filter(d => d.filial === 'N/A').forEach(d => { if(d.id_rota) rotasNaoEncontradas.add(d.id_rota) });
    arrayPenalidades.filter(p => p.filial === 'N/A').forEach(p => { if(p.id_rota) rotasNaoEncontradas.add(p.id_rota) });

    setPreviewBilling({
      diarias: arrayDiarias,
      penalidades: arrayPenalidades,
      filiaisNaoEncontradas: arrayDiarias.filter(d => d.filial === 'N/A').length + arrayPenalidades.filter(p => p.filial === 'N/A').length,
      rotasNaoEncontradas: Array.from(rotasNaoEncontradas)
    });

    setIsProcessingBilling(false);
  };

  const handleSaveBilling = async () => {
    setIsSavingBilling(true);
    addLog('[ETAPA 2] Salvando Billing no Firebase...', 'info');
    try {
      if (onImportBilling) {
        await onImportBilling(previewBilling.diarias, previewBilling.penalidades);
      }
      addLog('[ETAPA 2] Billing salvo com sucesso!', 'success');
      setPreviewBilling(null);
      setBillingFile(null);
      setQuinzenaBilling('');
      setNumeroFaturaBilling('');
    } catch (err) {
      addLog(`[ETAPA 2] Erro ao salvar: ${err.message}`, 'error');
    } finally {
      setIsSavingBilling(false);
    }
  };

  // ============================================================================
  // ETAPA 3: PROCESSAR CAP
  // ============================================================================
  const handleProcessCapCar = () => {
    if (!capcarFile) return;
    setIsProcessingCapCar(true);
    setLogs([]);
    setPreviewCapCar(null);
    addLog('[ETAPA 3] Iniciando processamento do CAP...', 'info');

    const processCapCarData = (dataArray) => {
      let headerRowIdx = -1;
      for (let i = 0; i < Math.min(15, dataArray.length); i++) {
        const row = dataArray[i];
        if (!row) continue;
        if (row.length > 3 && row.some(c => String(c).toUpperCase().includes('VENCIMENTO') || String(c).toUpperCase().includes('VALOR'))) {
          headerRowIdx = i;
          break;
        }
      }

      if (headerRowIdx === -1) {
        addLog('Erro: Não encontrou cabeçalhos conhecidos (VENCIMENTO, VALOR) no CAP.', 'error');
        setIsProcessingCapCar(false);
        return;
      }

      const headers = dataArray[headerRowIdx];
      const dataRows = dataArray.slice(headerRowIdx + 1);
      
      const parseCurrencyString = (str) => {
        let precoStr = String(str).toUpperCase().replace('R$', '').trim();
        if (precoStr.includes('.') && precoStr.includes(',')) {
          let lastDot = precoStr.lastIndexOf('.');
          let lastComma = precoStr.lastIndexOf(',');
          if (lastComma > lastDot) precoStr = precoStr.replace(/\./g, '').replace(',', '.');
          else precoStr = precoStr.replace(/,/g, '');
        } else if (precoStr.includes(',')) {
          precoStr = precoStr.replace(',', '.');
        }
        return parseFloat(precoStr) || 0;
      };

      const quinzenaHeader = headers[4];

      const enrichedData = dataRows.map(rowArray => {
        const newRow = {};
        headers.forEach((header, idx) => {
          if (!header) return;
          let val = rowArray[idx] !== undefined ? rowArray[idx] : '';
          if (typeof val === 'string') {
            const valUpper = val.toUpperCase();
            if (valUpper.includes('R$')) {
              newRow[header] = parseCurrencyString(val);
            } else if (String(header).toUpperCase().includes('AGREGADO') || String(header).toUpperCase().includes('MOTORISTA')) {
              newRow[header] = valUpper.trim();
            } else {
              newRow[header] = val.trim();
            }
          } else {
            newRow[header] = val;
          }
        });
        if (quinzenaHeader && newRow[quinzenaHeader]) {
          newRow.quinzena = newRow[quinzenaHeader];
        }
        return newRow;
      }).filter(r => Object.keys(r).length > 0 && r.quinzena);

      const quinzenasSet = new Set();
      enrichedData.forEach(r => { if (r.quinzena) quinzenasSet.add(r.quinzena); });

      setPreviewCapCar({
        dados: enrichedData,
        totalLinhas: enrichedData.length,
        quinzenas: Array.from(quinzenasSet)
      });
      setIsProcessingCapCar(false);
    };

    const ext = capcarFile.name.split('.').pop().toLowerCase();
    if (ext === 'csv') {
      Papa.parse(capcarFile, {
        skipEmptyLines: true,
        complete: (results) => processCapCarData(results.data),
        error: (err) => {
          addLog(`Erro CSV: ${err.message}`, 'error');
          setIsProcessingCapCar(false);
        }
      });
    } else {
      const reader = new FileReader();
      reader.onload = (e) => {
        try {
          const wb = XLSX.read(new Uint8Array(e.target.result), { type: 'array' });
          const ws = wb.Sheets[wb.SheetNames[0]];
          const dataArray = XLSX.utils.sheet_to_json(ws, { header: 1 });
          processCapCarData(dataArray);
        } catch (err) {
          addLog(`Erro XLSX: ${err.message}`, 'error');
          setIsProcessingCapCar(false);
        }
      };
      reader.readAsArrayBuffer(capcarFile);
    }
  };

  const handleSaveCapCar = async () => {
    setIsSavingCapCar(true);
    try {
      if (onImportCapCar) await onImportCapCar(previewCapCar.dados);
      addLog('[ETAPA 3] CAP salvo com sucesso!', 'success');
      setPreviewCapCar(null);
      setCapcarFile(null);
    } catch (err) {
      addLog(`[ETAPA 3] Erro ao salvar: ${err.message}`, 'error');
    } finally {
      setIsSavingCapCar(false);
    }
  };

  // ============================================================================
  // ETAPA 4: PROCESSAR OPERACIONAL BSC
  // ============================================================================
  const handleProcessBsc = () => {
    if (!bscFile) return;
    setIsProcessingBsc(true);
    setLogs([]);
    setPreviewBsc(null);
    addLog('[ETAPA 4] Iniciando processamento do Operacional BSC...', 'info');

    const processBscData = (dataArray) => {
      if (!dataArray || dataArray.length < 2) {
        addLog('Erro: Planilha BSC vazia ou sem dados.', 'error');
        setIsProcessingBsc(false);
        return;
      }

      let headerRowIdx = -1;
      for (let i = 0; i < Math.min(10, dataArray.length); i++) {
        const row = dataArray[i];
        if (!row) continue;
        if (row.some(c => String(c).toUpperCase() === 'ROTAS' || String(c).toUpperCase() === 'DRIVER')) {
          headerRowIdx = i;
          break;
        }
      }

      if (headerRowIdx === -1) {
        addLog('Erro: Não encontrou cabeçalho padrão BSC (ROTAS, DRIVER) na planilha.', 'error');
        setIsProcessingBsc(false);
        return;
      }

      const headers = dataArray[headerRowIdx].map(h => String(h || '').trim());
      
      const colData = headers.findIndex(h => h.toUpperCase() === 'DATA');
      const colRotas = headers.findIndex(h => h.toUpperCase() === 'ROTAS');
      const colDriver = headers.findIndex(h => h.toUpperCase() === 'DRIVER');
      const colAtStation = headers.findIndex(h => h.toUpperCase() === 'AT_STATION');
      
      if (colData === -1 || colRotas === -1 || colAtStation === -1) {
        addLog('Erro: Não encontrou as colunas DATA, ROTAS ou AT_STATION na planilha BSC.', 'error');
        setIsProcessingBsc(false);
        return;
      }

      // Mapeamentos para PROCV
      const mapRotaToFilial = {};
      const configMap = {};
      mapeamentoFiliais.forEach(m => configMap[String(m.filial).toUpperCase()] = m);

      // Auxiliares de mapeamento extraídos do próprio BD
      const mapDriverToMotorista = {};
      const mapRotaToMotorista = {};

      const populateMaps = (d) => {
        if (d.id_rota && d.filial && d.filial !== 'N/A') {
          const fKey = String(d.filial).toUpperCase();
          const config = configMap[fKey] || {};
          mapRotaToFilial[String(d.id_rota).trim()] = {
            filial: String(d.filial).trim(),
            regional: config.regional || 'N/A',
            supervisor: config.supervisor || 'N/A'
          };
        }
        if (d.driver_id && d.motorista && d.motorista !== 'N/A') {
          mapDriverToMotorista[String(d.driver_id).trim()] = d.motorista;
        }
        if (d.id_rota && d.motorista && d.motorista !== 'N/A') {
          mapRotaToMotorista[String(d.id_rota).trim()] = d.motorista;
        }
      };

      rawOperacionalData.forEach(populateMaps);
      rawFaturamentoData.forEach(populateMaps);

      const mapInsucessosRenomeados = {
        'INACCESSIBLE_ADDRESS': 'Area Inacessivel',
        'BUYER_REJECTED': 'Recusa',
        'BUYER_MOVED': 'Mudou Endereco',
        'BUYER_ABSENT': 'Cliente Ausente',
        'BUSINESS_CLOSED': 'Comercial',
        'BLOCKED_BY_KEYWORD': 'Palavra Chave',
        'BAD_ADDRESS': 'Nao Localizado',
        'NOT_ATTEMPTED': 'Nao Visitado',
        'TOTAL DE INSUCESSOS': 'Insucessos'
      };

      const translateHeader = (headerRaw) => {
        const hUpper = headerRaw.toUpperCase();
        for (const [key, val] of Object.entries(mapInsucessosRenomeados)) {
          if (hUpper.includes(key)) return val;
        }
        return headerRaw;
      };

      const diasDaSemanaMap = ['Dom', 'Seg', 'Ter', 'Qua', 'Qui', 'Sex', 'Sáb'];

      const newBscData = [];
      const quinzenasSet = new Set();
      let linhasProcessadas = 0;

      for (let i = headerRowIdx + 1; i < dataArray.length; i++) {
        const row = dataArray[i];
        if (!row || row.length < colRotas) continue;
        
        const dataRaw = String(row[colData] || '').trim();
        const idRota = String(row[colRotas] || '').trim();
        const driverId = colDriver !== -1 ? String(row[colDriver] || '').trim() : '';

        if (!dataRaw || !idRota) continue;

        let dataObj = new Date();
        const matchData = dataRaw.match(/(\d+)\s+de\s+([a-z]+)\.?\s+de\s+(\d+)/i);
        if (matchData) {
          const dia = parseInt(matchData[1]);
          const mesNome = matchData[2].toLowerCase();
          const ano = parseInt(matchData[3]);
          const meses = { jan:0, fev:1, mar:2, abr:3, mai:4, jun:5, jul:6, ago:7, set:8, out:9, nov:10, dez:11 };
          const mes = meses[mesNome.substring(0,3)] || 0;
          dataObj = new Date(ano, mes, dia);
        } else if (dataRaw.includes('/')) {
          const parts = dataRaw.split('/');
          if (parts.length === 3) dataObj = new Date(parts[2], parts[1]-1, parts[0]);
        }

        const diaNum = dataObj.getDate();
        const mesNum = dataObj.getMonth() + 1;
        const anoNum = dataObj.getFullYear();
        
        const mesStr = mesNum.toString().padStart(2, '0');
        const qIndicator = diaNum <= 15 ? 'Q1' : 'Q2';
        const quinzenaGerada = `${anoNum}${mesStr}${qIndicator}`;
        quinzenasSet.add(quinzenaGerada);

        const dataFormatada = `${diaNum.toString().padStart(2, '0')}/${mesStr}/${anoNum}`;
        const diaSemanaStr = diasDaSemanaMap[dataObj.getDay()] || 'N/A';

        const routeMapObj = mapRotaToFilial[idRota] || { filial: 'N/A', regional: 'N/A', supervisor: 'N/A' };
        
        let motoristaEncontrado = 'N/A';
        if (driverId && mapDriverToMotorista[driverId]) {
          motoristaEncontrado = mapDriverToMotorista[driverId];
        } else if (mapRotaToMotorista[idRota]) {
          motoristaEncontrado = mapRotaToMotorista[idRota];
        }

        const baseBscRow = {
          quinzena: quinzenaGerada,
          data: dataFormatada,
          dia_semana: diaSemanaStr,
          id_rota: idRota,
          driver_id: driverId,
          motorista: motoristaEncontrado,
          filial: routeMapObj.filial,
          regional: routeMapObj.regional,
          supervisor: routeMapObj.supervisor
        };

        headers.forEach((h, idx) => {
          if (idx >= colAtStation) {
            const renomeado = translateHeader(h);
            const valNum = parseFloat(row[idx]) || 0;
            baseBscRow[renomeado] = valNum;
          } else {
            const kName = h.toLowerCase().replace(/[\s\.]/g, '_');
            if (h && !['DATA', 'ROTAS', 'DRIVER'].includes(h.toUpperCase())) {
               if (!baseBscRow.hasOwnProperty(kName)) {
                 const v = row[idx];
                 baseBscRow[kName] = isNaN(v) ? String(v).trim() : parseFloat(v);
               }
            }
          }
        });

        newBscData.push(baseBscRow);
        linhasProcessadas++;
      }

      const quinzenasAchadas = Array.from(quinzenasSet);
      addLog(`Dados BSC processados: ${linhasProcessadas} linhas. Quinzenas: ${quinzenasAchadas.join(', ')}`, 'success');

      setPreviewBsc({
        dados: newBscData,
        quinzenas: quinzenasAchadas
      });
      setIsProcessingBsc(false);
    };

    const ext = bscFile.name.split('.').pop().toLowerCase();
    if (ext === 'csv') {
      Papa.parse(bscFile, {
        skipEmptyLines: true,
        complete: (results) => processBscData(results.data),
        error: (err) => {
          addLog(`Erro CSV BSC: ${err.message}`, 'error');
          setIsProcessingBsc(false);
        }
      });
    } else {
      const reader = new FileReader();
      reader.onload = (e) => {
        try {
          const wb = XLSX.read(new Uint8Array(e.target.result), { type: 'array' });
          const ws = wb.Sheets[wb.SheetNames[0]];
          const dataArray = XLSX.utils.sheet_to_json(ws, { header: 1 });
          processBscData(dataArray);
        } catch (err) {
          addLog(`Erro XLSX BSC: ${err.message}`, 'error');
          setIsProcessingBsc(false);
        }
      };
      reader.readAsArrayBuffer(bscFile);
    }
  };

  const handleSaveBsc = async () => {
    setIsSavingBsc(true);
    addLog('[ETAPA 4] Salvando Operacional BSC no Firebase...', 'info');
    try {
      if (onImportOperacionalBSC) {
        await onImportOperacionalBSC(previewBsc.dados);
      }
      addLog('[ETAPA 4] BSC salvo com sucesso!', 'success');
      setPreviewBsc(null);
      setBscFile(null);
    } catch (err) {
      addLog(`[ETAPA 4] Erro ao salvar BSC: ${err.message}`, 'error');
    } finally {
      setIsSavingBsc(false);
    }
  };

  // ============================================================================
  // RENDER
  // ============================================================================
  return (
    <div className="w-full max-w-5xl mx-auto flex flex-col gap-6">
      
      <div className="bg-white rounded-3xl p-8 shadow-sm border border-slate-200">
        <div className="flex items-start gap-4 mb-4">
          <div className="bg-blue-100 text-blue-600 p-3 rounded-2xl">
            <Database className="w-8 h-8" />
          </div>
          <div>
            <h1 className="text-2xl font-black text-slate-800">Importador Inteligente (ETL)</h1>
            <p className="text-slate-500 font-medium">Importe em etapas. O cruzamento usa toda a base do Firebase.</p>
          </div>
        </div>

        <div className="flex gap-2 mt-4 flex-wrap">
          <button
            onClick={() => setActiveStep(1)}
            className={`flex flex-col items-center p-4 rounded-xl border-2 transition-all duration-300 relative overflow-hidden group ${
              activeStep === 1
                ? 'border-blue-500 bg-blue-500/10 text-blue-600'
                : previewOp 
                  ? 'border-green-500/50 bg-green-500/5 text-green-600 hover:border-green-400'
                  : 'border-slate-200 hover:border-slate-300 hover:bg-slate-50 text-slate-400'
            }`}
          >
            <div className={`p-3 rounded-xl mb-3 ${activeStep === 1 ? 'bg-blue-500/20' : previewOp ? 'bg-green-500/20' : 'bg-slate-100'}`}>
              <UploadCloud size={24} className={activeStep === 1 ? 'text-blue-600' : previewOp ? 'text-green-600' : 'text-slate-400'} />
            </div>
            <span className="font-semibold text-center">Etapa 1:<br/>Operacional</span>
          </button>

          <button
            onClick={() => setActiveStep(2)}
            className={`flex flex-col items-center p-4 rounded-xl border-2 transition-all duration-300 relative overflow-hidden group ${
              activeStep === 2
                ? 'border-emerald-500 bg-emerald-500/10 text-emerald-600'
                : previewBilling 
                  ? 'border-green-500/50 bg-green-500/5 text-green-600 hover:border-green-400'
                  : 'border-slate-200 hover:border-slate-300 hover:bg-slate-50 text-slate-400'
            }`}
          >
            <div className={`p-3 rounded-xl mb-3 ${activeStep === 2 ? 'bg-emerald-500/20' : previewBilling ? 'bg-green-500/20' : 'bg-slate-100'}`}>
              <FileSpreadsheet size={24} className={activeStep === 2 ? 'text-emerald-600' : previewBilling ? 'text-green-600' : 'text-slate-400'} />
            </div>
            <span className="font-semibold text-center">Etapa 2:<br/>Billing</span>
          </button>

          <button
            onClick={() => setActiveStep(3)}
            className={`flex flex-col items-center p-4 rounded-xl border-2 transition-all duration-300 relative overflow-hidden group ${
              activeStep === 3
                ? 'border-purple-500 bg-purple-500/10 text-purple-600'
                : previewCapCar 
                  ? 'border-green-500/50 bg-green-500/5 text-green-600 hover:border-green-400'
                  : 'border-slate-200 hover:border-slate-300 hover:bg-slate-50 text-slate-400'
            }`}
          >
            <div className={`p-3 rounded-xl mb-3 ${activeStep === 3 ? 'bg-purple-500/20' : previewCapCar ? 'bg-green-500/20' : 'bg-slate-100'}`}>
              <DollarSign size={24} className={activeStep === 3 ? 'text-purple-600' : previewCapCar ? 'text-green-600' : 'text-slate-400'} />
            </div>
            <span className="font-semibold text-center">Etapa 3:<br/>CAP</span>
          </button>

          <button
            onClick={() => setActiveStep(4)}
            className={`flex flex-col items-center p-4 rounded-xl border-2 transition-all duration-300 relative overflow-hidden group ${
              activeStep === 4
                ? 'border-indigo-500 bg-indigo-500/10 text-indigo-600'
                : previewBsc 
                  ? 'border-green-500/50 bg-green-500/5 text-green-600 hover:border-green-400'
                  : 'border-slate-200 hover:border-slate-300 hover:bg-slate-50 text-slate-400'
            }`}
          >
            <div className={`p-3 rounded-xl mb-3 ${activeStep === 4 ? 'bg-indigo-500/20' : previewBsc ? 'bg-green-500/20' : 'bg-slate-100'}`}>
              <Box size={24} className={activeStep === 4 ? 'text-indigo-600' : previewBsc ? 'text-green-600' : 'text-slate-400'} />
            </div>
            <span className="font-semibold text-center">Etapa 4:<br/>Operacional BSC</span>
          </button>
        </div>
      </div>

      {activeStep === 1 && (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col justify-center relative overflow-hidden">
            <div className="relative z-10 flex flex-col h-full justify-between">
              <div>
                <div className="flex items-center gap-3 mb-6">
                  <div className="w-12 h-12 bg-blue-100 rounded-2xl flex items-center justify-center">
                    <UploadCloud className="w-6 h-6 text-blue-600" />
                  </div>
                  <div>
                    <h2 className="text-xl font-bold text-slate-800">Upload do Operacional</h2>
                  </div>
                </div>

                <div className="bg-slate-50 border border-slate-200 rounded-2xl p-6 mb-6">
                  <input type="file" accept=".csv, .xlsx, .xls" className="hidden" id="fileOp" onChange={(e) => setBaseFile(e.target.files[0])} />
                  <label htmlFor="fileOp" className="cursor-pointer flex flex-col items-center gap-4 text-center">
                    <UploadCloud className="w-8 h-8 text-slate-400" />
                    <p className="text-slate-700 font-medium">{baseFile ? baseFile.name : 'Selecionar arquivo Operacional'}</p>
                  </label>
                </div>

                <div className="w-full text-left mb-6">
                  <label className="block text-xs font-bold text-slate-500 uppercase tracking-wider mb-2">Quinzena do Operacional *</label>
                  <input
                    type="text"
                    value={quinzenaOp}
                    onChange={(e) => setQuinzenaOp(e.target.value)}
                    placeholder="Ex: 202606Q1"
                    className="w-full bg-white border border-slate-300 rounded-xl px-4 py-3 text-slate-800 font-bold focus:outline-none focus:ring-2 focus:ring-blue-500/20 focus:border-blue-500 transition-all"
                  />
                </div>

                {baseFile && !previewOp && (
                  <button onClick={handleProcessOperacional} disabled={isProcessingOp || !quinzenaOp.trim()} className="w-full bg-blue-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                    {isProcessingOp ? 'Processando...' : 'Processar Planilha'}
                  </button>
                )}
              </div>
            </div>
          </div>

          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col">
            <h2 className="text-lg font-bold text-slate-800 mb-6">Preview e Envio (Etapa 1)</h2>
            {previewOp ? (
              <div className="flex-1 flex flex-col bg-slate-50 border border-slate-200 rounded-2xl p-6">
                <p className="text-green-600 font-bold mb-4">Processamento Concluído!</p>
                <button onClick={handleSaveOperacional} disabled={isSavingOp} className="w-full mt-auto bg-green-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                  {isSavingOp ? 'Sincronizando...' : 'Sincronizar com o Firebase'}
                </button>
              </div>
            ) : (
              <div className="flex-1 flex flex-col items-center justify-center bg-slate-50 border border-slate-200 rounded-2xl border-dashed p-6 text-center">
                <p className="text-slate-500">Faça o upload e processe para ver os detalhes da importação.</p>
              </div>
            )}
          </div>
        </div>
      )}

      {activeStep === 2 && (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col justify-center relative overflow-hidden">
            <div className="relative z-10 flex flex-col h-full justify-between">
              <div>
                <div className="flex items-center gap-3 mb-6">
                  <div className="w-12 h-12 bg-emerald-100 rounded-2xl flex items-center justify-center">
                    <FileSpreadsheet className="w-6 h-6 text-emerald-600" />
                  </div>
                  <div>
                    <h2 className="text-xl font-bold text-slate-800">Upload do Billing</h2>
                  </div>
                </div>

                <div className="bg-slate-50 border border-slate-200 rounded-2xl p-6 mb-6">
                  <input type="file" accept=".csv, .xlsx, .xls" className="hidden" id="fileBilling" onChange={(e) => setBillingFile(e.target.files[0])} />
                  <label htmlFor="fileBilling" className="cursor-pointer flex flex-col items-center gap-4 text-center">
                    <UploadCloud className="w-8 h-8 text-slate-400" />
                    <p className="text-slate-700 font-medium">{billingFile ? billingFile.name : 'Selecionar arquivo Billing'}</p>
                  </label>
                </div>

                <div className="w-full flex gap-4 text-left mb-6">
                  <div className="flex-1">
                    <label className="block text-xs font-bold text-slate-500 uppercase tracking-wider mb-2">Quinzena</label>
                    <input
                      type="text"
                      value={quinzenaBilling}
                      onChange={(e) => setQuinzenaBilling(e.target.value)}
                      placeholder="Ex: 202606Q1"
                      className="w-full bg-white border border-slate-300 rounded-xl px-4 py-3 text-slate-800 font-bold focus:outline-none focus:ring-2 focus:ring-emerald-500/20 focus:border-emerald-500 transition-all"
                    />
                  </div>
                  <div className="flex-1">
                    <label className="block text-xs font-bold text-slate-500 uppercase tracking-wider mb-2">NF</label>
                    <input
                      type="text"
                      value={numeroFaturaBilling}
                      onChange={(e) => setNumeroFaturaBilling(e.target.value)}
                      placeholder="Ex: FAT-123"
                      className="w-full bg-white border border-slate-300 rounded-xl px-4 py-3 text-slate-800 font-bold focus:outline-none focus:ring-2 focus:ring-emerald-500/20 focus:border-emerald-500 transition-all"
                    />
                  </div>
                </div>

                {billingFile && !previewBilling && (
                  <button onClick={handleProcessBilling} disabled={isProcessingBilling} className="w-full bg-emerald-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                    {isProcessingBilling ? 'Processando...' : 'Processar Billing'}
                  </button>
                )}
              </div>
            </div>
          </div>

          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col">
            <h2 className="text-lg font-bold text-slate-800 mb-6">Preview e Envio (Etapa 2)</h2>
            {previewBilling ? (
              <div className="flex-1 flex flex-col bg-slate-50 border border-slate-200 rounded-2xl p-6">
                <p className="text-green-600 font-bold mb-4">Processamento Concluído!</p>
                <div className="grid grid-cols-2 gap-4 mb-8">
                  <div className="bg-white p-4 rounded-xl border border-slate-200 text-center">
                    <p className="text-slate-500 text-xs font-bold uppercase tracking-wider mb-1">Diárias</p>
                    <p className="text-2xl font-black text-slate-800">{previewBilling.diarias.length}</p>
                  </div>
                  <div className="bg-white p-4 rounded-xl border border-slate-200 text-center">
                    <p className="text-slate-500 text-xs font-bold uppercase tracking-wider mb-1">Penalidades</p>
                    <p className="text-2xl font-black text-slate-800">{previewBilling.penalidades.length}</p>
                  </div>
                </div>
                <button onClick={handleSaveBilling} disabled={isSavingBilling} className="w-full mt-auto bg-green-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                  {isSavingBilling ? 'Sincronizando...' : 'Sincronizar com o Firebase'}
                </button>
              </div>
            ) : (
              <div className="flex-1 flex flex-col items-center justify-center bg-slate-50 border border-slate-200 rounded-2xl border-dashed p-6 text-center">
                <p className="text-slate-500">Faça o upload e processe para ver os detalhes da importação.</p>
              </div>
            )}
          </div>
        </div>
      )}

      {activeStep === 3 && (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col justify-center relative overflow-hidden">
            <div className="relative z-10 flex flex-col h-full justify-between">
              <div>
                <div className="flex items-center gap-3 mb-6">
                  <div className="w-12 h-12 bg-purple-100 rounded-2xl flex items-center justify-center">
                    <DollarSign className="w-6 h-6 text-purple-600" />
                  </div>
                  <div>
                    <h2 className="text-xl font-bold text-slate-800">Upload do CAP</h2>
                  </div>
                </div>

                <div className="bg-slate-50 border border-slate-200 rounded-2xl p-6 mb-6">
                  <input type="file" accept=".csv, .xlsx, .xls" className="hidden" id="fileCap" onChange={(e) => setCapcarFile(e.target.files[0])} />
                  <label htmlFor="fileCap" className="cursor-pointer flex flex-col items-center gap-4 text-center">
                    <UploadCloud className="w-8 h-8 text-slate-400" />
                    <p className="text-slate-700 font-medium">{capcarFile ? capcarFile.name : 'Selecionar arquivo CAP'}</p>
                  </label>
                </div>

                {capcarFile && !previewCapCar && (
                  <button onClick={handleProcessCapCar} disabled={isProcessingCapCar} className="w-full bg-purple-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                    {isProcessingCapCar ? 'Processando...' : 'Processar CAP'}
                  </button>
                )}
              </div>
            </div>
          </div>

          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col">
            <h2 className="text-lg font-bold text-slate-800 mb-6">Preview e Envio (Etapa 3)</h2>
            {previewCapCar ? (
              <div className="flex-1 flex flex-col bg-slate-50 border border-slate-200 rounded-2xl p-6">
                <p className="text-green-600 font-bold mb-4">Processamento Concluído!</p>
                <button onClick={handleSaveCapCar} disabled={isSavingCapCar} className="w-full mt-auto bg-green-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                  {isSavingCapCar ? 'Sincronizando...' : 'Sincronizar com o Firebase'}
                </button>
              </div>
            ) : (
              <div className="flex-1 flex flex-col items-center justify-center bg-slate-50 border border-slate-200 rounded-2xl border-dashed p-6 text-center">
                <p className="text-slate-500">Faça o upload e processe para ver os detalhes da importação.</p>
              </div>
            )}
          </div>
        </div>
      )}

      {activeStep === 4 && (
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8">
          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col justify-center relative overflow-hidden">
            <div className="relative z-10 flex flex-col h-full justify-between">
              <div>
                <div className="flex items-center gap-3 mb-6">
                  <div className="w-12 h-12 bg-indigo-100 rounded-2xl flex items-center justify-center">
                    <Box className="w-6 h-6 text-indigo-600" />
                  </div>
                  <div>
                    <h2 className="text-xl font-bold text-slate-800">Upload do Operacional BSC</h2>
                  </div>
                </div>

                <div className="bg-slate-50 border border-slate-200 rounded-2xl p-6 mb-6">
                  <input type="file" accept=".csv, .xlsx, .xls" className="hidden" id="fileBsc" onChange={(e) => setBscFile(e.target.files[0])} />
                  <label htmlFor="fileBsc" className="cursor-pointer flex flex-col items-center gap-4 text-center">
                    <UploadCloud className="w-8 h-8 text-slate-400" />
                    <p className="text-slate-700 font-medium">{bscFile ? bscFile.name : 'Selecionar arquivo BSC'}</p>
                  </label>
                </div>

                {bscFile && !previewBsc && (
                  <button onClick={handleProcessBsc} disabled={isProcessingBsc} className="w-full bg-indigo-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                    {isProcessingBsc ? 'Processando...' : 'Processar Planilha BSC'}
                  </button>
                )}
              </div>
            </div>
          </div>

          <div className="bg-white p-6 md:p-8 rounded-3xl border border-slate-200 shadow-xl flex flex-col">
            <h2 className="text-lg font-bold text-slate-800 mb-6">Preview e Envio (Etapa 4)</h2>
            {previewBsc ? (
              <div className="flex-1 flex flex-col bg-slate-50 border border-slate-200 rounded-2xl p-6">
                <p className="text-green-600 font-bold mb-4">Processamento BSC Concluído!</p>
                <div className="grid grid-cols-2 gap-4 mb-8">
                  <div className="bg-white p-4 rounded-xl border border-slate-200 text-center">
                    <p className="text-slate-500 text-xs font-bold uppercase tracking-wider mb-1">Linhas Válidas</p>
                    <p className="text-2xl font-black text-slate-800">{previewBsc.dados.length}</p>
                  </div>
                  <div className="bg-white p-4 rounded-xl border border-slate-200 text-center">
                    <p className="text-slate-500 text-xs font-bold uppercase tracking-wider mb-1">Quinzenas Detectadas</p>
                    <p className="text-sm font-bold text-indigo-600">
                      {previewBsc.quinzenas.length > 0 ? previewBsc.quinzenas.join(', ') : 'Nenhuma'}
                    </p>
                  </div>
                </div>
                <button onClick={handleSaveBsc} disabled={isSavingBsc} className="w-full mt-auto bg-green-600 text-white font-bold py-4 rounded-xl disabled:opacity-50">
                  {isSavingBsc ? 'Sincronizando...' : 'Sincronizar com o Firebase'}
                </button>
              </div>
            ) : (
              <div className="flex-1 flex flex-col items-center justify-center bg-slate-50 border border-slate-200 rounded-2xl border-dashed p-6 text-center">
                <p className="text-slate-500">Faça o upload e processe para ver os detalhes da importação.</p>
              </div>
            )}
          </div>
        </div>
      )}

      {logs.length > 0 && (
        <div className="bg-slate-900 rounded-3xl p-6 shadow-xl">
          <h3 className="text-white font-bold mb-4 flex items-center gap-2">Console de Processamento</h3>
          <div className="bg-slate-950 p-4 rounded-xl font-mono text-sm h-48 overflow-y-auto">
            {logs.map((log, idx) => (
              <div key={idx} className={log.type === 'error' ? 'text-red-400' : 'text-slate-400'}>{log.msg}</div>
            ))}
          </div>
        </div>
      )}
    </div>
  );
}
